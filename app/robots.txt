import { MetadataRoute } from 'next';

export default function robots(): MetadataRoute.Robots {
  const baseUrl = 'https://factu.me';

  return {
    rules: [
      {
        userAgent: '*',
        allow: '/',
        disallow: ['/app/', '/api/', '/share/', '/sign/', '/client/'],
      },
    ],
    sitemap: `${baseUrl}/sitemap.xml`,
  };
}
